Class Grid

Creates a grid view of a sql statement

Usage

$dbg = new \Module\DB\Grid('users', 'SELECT * FROM users');

$dbg->exclude(array(
  'user_id',
  'password'
));

$dbg->mapFields(array(
  'name' => 'user_id'
));

$dbg->sortable(array(
  'name',
  'date'
));

$dbg->show();
