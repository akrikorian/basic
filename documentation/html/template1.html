Template Class

Basic templating functionality

Usage:

index.php

require_once('include/sleepy.php');
$page = new Template('templates/default.tpl');
$page->bind('title', 'Sleepy Mustache');
$page->bind('header', 'Hello world!');
$page->show();

default.tpl

<html>
<head>
<title>{{ title }}</title>
</head>
<body>
<h1>{{ header }}</h1>
<p>This page has been viewed {{ hits }} times.</p>
</body>
</html>
